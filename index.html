    # New Compute Function
    @api.depends('order_line.margin', 'amount_untaxed')
    def _compute_business_revenue_margin(self):
        for order in self:
            # LICENSE
            order.margin_license = sum(order.order_line.filtered(lambda x: x.business_revenue == 'license').mapped('margin'))
            untaxed = sum(order.order_line.filtered(lambda x: x.business_revenue == 'license').mapped("price_subtotal"))
            order.margin_license_percent = untaxed and order.margin_license/untaxed
        
            # IMPLEMENTATION
            order.margin_imple = sum(order.order_line.filtered(lambda x: x.business_revenue == 'implementation').mapped('margin'))
            untaxed = sum(order.order_line.filtered(lambda x: x.business_revenue == 'implementation').mapped("price_subtotal"))
            order.margin_imple_percent = untaxed and order.margin_imple/untaxed
        
            # MAINTENANCE
            order.margin_maintenance = sum(order.order_line.filtered(lambda x: x.business_revenue == 'maintenance').mapped('margin'))
            untaxed = sum(order.order_line.filtered(lambda x: x.business_revenue == 'maintenance').mapped("price_subtotal"))
            order.margin_maintenance_percent = untaxed and order.margin_maintenance/untaxed

    @api.depends('order_line.margin', 'amount_untaxed', 'cost_amount')
    def _compute_margin(self):
        if not all(self._ids):
            for order in self:
                order.margin = order.amount_untaxed - order.cost_amount # originalnya -> sum(order.order_line.mapped('margin'))
                order.margin_percent = order.amount_untaxed and order.margin/order.amount_untaxed
        else:
            grouped_order_lines_data = self.env['sale.order.line']._read_group(
                [
                    ('order_id', 'in', self.ids),
                ], ['order_id'], ['margin:sum'])
            mapped_data = {order.id: margin for order, margin in grouped_order_lines_data}
            for order in self:
                order.margin = order.amount_untaxed - order.cost_amount # originalnya -> mapped_data.get(order.id, 0.0)
                order.margin_percent = order.amount_untaxed and order.margin/order.amount_untaxed

    def create_budget_project(self):

class RevaluasiKurs(models.TransientModel):
    _name = 'revaluasi.kurs'

    ref = fields.Char(string="Reference")

    date = fields.Date('Date', default=fields.Date.context_today, required=True,)

    def _defaultAccount(self):
        dtAcc = self.env['account.account'].search([('user_type_id.name','in',MAP_ACCOUNT_REVAL)])
        return dtAcc.ids
    account_ids = fields.Many2many('account.account', string='Account', domain=[('user_type_id.name', 'in', MAP_ACCOUNT_REVAL)], default=False)
    
    def _defaultCurrency(self):
        line_curr = []
        dt = self.env['res.currency'].search([('active','=',True),('id','!=',self.env.company.currency_id.id)])
        dtAcc = self.env['account.account'].search([('user_type_id.name','in',MAP_ACCOUNT_REVAL)])
        if dt:
            for i in dt:
                tmp = [0,0,{
                    'currency_id': i.id,
                    'rate_kurs': self.env['res.currency.rate'].search([('currency_id','=',i.id),('name','=',datetime.date.today())]).inverse_company_rate or 0
                }]
                line_curr.append(tmp)
        return line_curr
    revaluasi_line = fields.One2many('revaluasi.kurs.line','revaluasi_id', string="Lines", ondelete='cascade', copy=True, tracking=True, default=_defaultCurrency)
    
    @api.onchange('date')
    def _onchange_date(self):
        for i in self.revaluasi_line:
            i._onchange_kurs()
    
    def action_confirm(self):
        result = []
        # SET TGL AWAL DAN TGL AKHIR
        tgl_awal_bln = datetime.datetime(self.date.year, self.date.month, 1)
        day_of_month = calendar.monthrange(self.date.year, self.date.month)[1]
        tgl_akhir_bln = datetime.datetime(self.date.year, self.date.month, day_of_month)

        for i in self.revaluasi_line:
            curr = self.env['res.currency.rate'].search([('name','=',self.date),('currency_id','=',i.currency_id.id)])
            if curr:
                curr.write({'inverse_company_rate': i.rate_kurs})
            else:
                curr.create({
                    'currency_id': i.currency_id.id,
                    'inverse_company_rate': i.rate_kurs,
                    'name': self.date,
                })
            move_line = self.env['account.move.line'].search([
                '&', 
                ('date', '<=', tgl_akhir_bln), 
                ('account_id', 'not in', [i.realized_account_id.id, i.unrealized_account_id.id]),
                ('account_id.user_type_id.name','in',MAP_ACCOUNT_REVAL),
                ('account_id','in',self.account_ids.ids),
                ('amount_currency', '!=', False),('amount_currency', '!=', 0),
                ('currency_id','=', i.currency_id.id), 
                ('parent_state','=','posted'),
            ])
            map_coa = move_line.mapped('account_id')
            line_ids_det = []
            for i2 in map_coa:
                # SALDO
                line_saldo = self.env['account.move.line'].search([
                    '&',
                    ('date', '<=', tgl_akhir_bln), 
                    ('account_id', '=', i2.id),
                    ('currency_id', '=', i.currency_id.id),
                    ('amount_currency', '!=', False), ('amount_currency', '!=', 0),
                    ('parent_state', '=', 'posted'),
                ])

                # parsing SALDO
                debit_saldo = sum(line_saldo.mapped('debit'))
                credit_saldo = sum(line_saldo.mapped('credit'))
                amount_curr_saldo = sum(line_saldo.mapped('amount_currency'))

                balance = debit_saldo - credit_saldo
                balance_rate = i.currency_id._convert(amount_curr_saldo, self.env.company.currency_id, self.env.company, tgl_akhir_bln)
                balance_value = balance - balance_rate

                if balance_value != 0 and balance != 0:
                    move_line_current_month = self.env['account.move.line'].search([
                        '&',
                        ('date', '>=', tgl_awal_bln), ('date', '<=', tgl_akhir_bln), 
                        ('account_id.user_type_id.name','in',MAP_ACCOUNT_REVAL),
                        ('account_id','=',i2.id),
                        ('amount_currency', '!=', False),('amount_currency', '!=', 0),
                        ('currency_id','=', i.currency_id.id), 
                        ('parent_state','=','posted'),
                        ('partner_id','!=',False)
                    ])
                    acc_res = 0
                    # jika pada bulan tersebut ada transaksi, maka masuk ke akun realized
                    # jika pada bulan tersebut tidak ada transaksi, maka masuk ke akun unrealized
                    if len(move_line_current_month) > 0:
                        acc_res = i.realized_account_id.id
                    else:
                        acc_res = i.unrealized_account_id.id

                    # VALUE AFTER SELISIH RATE
                    det = {
                        'name': str(self.label) + ' ' + str(i.currency_id.name) + ' ' + tgl_akhir_bln.strftime('%B'),
                        'debit': balance_value < 0.0 and -balance_value or 0.0,
                        'credit': balance_value > 0.0 and balance_value or 0.0,
                        'date_maturity': tgl_akhir_bln.strftime('%Y-%m-%d'),
                        'account_id': i2.id,
                        'currency_id': i.currency_id.id, #self.env.company.currency_id.id,
                        'amount_currency': 0,
                    }
                    line_ids_det.append((0,0,det))

                    det = {
                        'name': str(self.label) + ' ' + str(i.currency_id.name) + ' ' + tgl_akhir_bln.strftime('%B'),
                        'debit': balance_value > 0.0 and balance_value or 0.0,
                        'credit': balance_value < 0.0 and -balance_value or 0.0,
                        'date_maturity': tgl_akhir_bln.strftime('%Y-%m-%d'),
                        'account_id': acc_res,
                        'currency_id': i.currency_id.id, #self.env.company.currency_id.id,
                        'amount_currency': 0,
                    }
                    line_ids_det.append((0,0,det))

            all_move_vals = {
                'date': tgl_akhir_bln.strftime('%Y-%m-%d'),
                'ref': str(self.ref) + ' ' + str(i.currency_id.name),
                'journal_id': self.journal_id.id,
                'line_ids': line_ids_det
            }

            # create account move
            if line_ids_det != []:
                am = self.env['account.move'].with_context(default_type='entry')
                moves = am.create(all_move_vals)
                result.append(moves.id)

        # =============================================
        # if result:
        open_form = {
            'name': _("Journal Revaluasi Kurs"),
            'type': 'ir.actions.act_window',
            'res_model': 'account.move',
            'view_mode': 'tree,form',
            # 'res_id': result,
            'domain': [('id', 'in', result)],
            'context': {'create': False},
        }
        return open_form

class addColumbariumWizard(models.TransientModel):
    _name = 'add.columbarium.wizard'
    
    sale_order_id = fields.Many2one('sale.order', string='Sale Order', required=True)
    currency_id = fields.Many2one(related="sale_order_id.currency_id")
    
    def default_pickup_date(self):
        # Ambil waktu lokal user
        user_today = fields.Datetime.context_timestamp(self, datetime.now()).date()
        dt = datetime.combine(user_today, time(hour=0, minute=5))
        # Kurangi 7 jam (backdate)
        return dt - timedelta(hours=7)

    def default_return_date_others(self):
        local_target_time = datetime.combine(
            datetime.now().date() + relativedelta(years=1, days=-1),
            time(hour=12, minute=5)
        )
        return local_target_time - timedelta(hours=7)

    def _get_product_columbarium_id_domain(self):
        columbarium_category_id = self.env.ref('custom_sale_order.categ_columbarium')
        return [('categ_id','=',columbarium_category_id.id)]
    
    product_columbarium_id = fields.Many2one('product.product', string='Columbarium',domain=_get_product_columbarium_id_domain)
    pickup_date = fields.Datetime(string="Tanggal Pickup",  default=default_pickup_date)
    return_date = fields.Datetime(string="Tanggal Return", compute='_compute_return_date', store=True)
    pickup_date_others = fields.Datetime(string="Tanggal Pickup",default=default_pickup_date)
    return_date_others = fields.Datetime(string="Tanggal Return", default=default_return_date_others,required=True)
    list_price = fields.Float(compute='_compute_list_price', string='Subtotal')
    duration = fields.Selection([
        ('1m', '1 Bulan'),
        ('3m', '3 Bulan'),
        ('6m', '6 Bulan'),
        ('1y', '1 Tahun'),
        ('5y', '5 Tahun'),
        ('10y', '10 Tahun')
    ], string='Durasi')
    use_other_date = fields.Boolean(string="Tanggal Lainnya")
    return_date = fields.Datetime(string="Tanggal Return", compute='_compute_return_date', store=True)
    
    # new field for status rental detail
    rental_status = fields.Selection([
        ('not_available', 'Not Available'),
        ('free', 'Available'),
        ('reserve', 'Reserved'),
        ('occupied', 'Occupied'),
    ], compute='_compute_rental_status', string='Status Rental')
    rental_status_detail = fields.Char(compute='_compute_rental_status', string='Status Rental Detail')
    @api.depends('product_columbarium_id','pickup_date','return_date','pickup_date_others','return_date_others','use_other_date')
    def _compute_rental_status(self):
        for record in self:
            record.rental_status = None
            record.rental_status_detail = None
            if record.product_columbarium_id and (record.pickup_date and record.return_date) or \
                (record.pickup_date_others and record.return_date_others):
                rental_status_json = record.product_columbarium_id.product_tmpl_id.get_rental_status_info(
                    product_id = record.product_columbarium_id,
                    start_date = record.pickup_date if not record.use_other_date else record.pickup_date_others,
                    end_date = record.return_date if not record.use_other_date else record.return_date_others,
                )
                if rental_status_json:
                    record.rental_status = rental_status_json.get('status')
                    record.rental_status_detail = "{status} {detail}".format(
                        status= dict(self._fields['rental_status'].selection).get(record.rental_status),
                        detail= rental_status_json.get('detail') if rental_status_json.get('detail') else ""
                    )

    @api.depends('pickup_date', 'duration')
    def _compute_return_date(self):
        user_tz = self.env.user.tz or 'UTC'
        local_tz = pytz.timezone(user_tz)
        utc_tz = pytz.utc

        for record in self:
            if record.pickup_date and record.duration:
                local_pickup_datetime = pytz.utc.localize(record.pickup_date).astimezone(local_tz)
                start_date_local = local_pickup_datetime.date() 

                calculated_date_local = False
                if 'm' in record.duration:
                    months = int(record.duration.replace('m', ''))
                    calculated_date_local = start_date_local + relativedelta(months=months) - timedelta(days=1)
                elif 'y' in record.duration:
                    years = int(record.duration.replace('y', ''))
                    calculated_date_local = start_date_local + relativedelta(years=years) - timedelta(days=1)
                
                if calculated_date_local:
                    return_datetime_local = local_tz.localize(datetime(
                        calculated_date_local.year,
                        calculated_date_local.month,
                        calculated_date_local.day,
                        12, 5, 0
                    ))
                    record.return_date = return_datetime_local.astimezone(utc_tz).replace(tzinfo=None)
                else:
                    record.return_date = False
            else:
                record.return_date = False

    @api.depends('product_columbarium_id', 'pickup_date', 'return_date', 'pickup_date_others', 'return_date_others', 'use_other_date')
    def _compute_list_price(self):
        for record in self:
            record.list_price = 0

            if record.use_other_date:
                start_date = record.pickup_date_others
                end_date = record.return_date_others
            else:
                start_date = record.pickup_date
                end_date = record.return_date

            if record.product_columbarium_id and start_date and end_date:
                record.list_price = record.product_columbarium_id.product_tmpl_id.get_rental_list_price(
                    product_id=record.product_columbarium_id,
                    start_date=start_date,
                    end_date=end_date + timedelta(days=1),
                )

    def action_add(self):
        for record in self:
            err_msg = ""
            if record.rental_status == 'occupied': 
                err_msg = "Ruang Columbarium tersebut telah disewa."
                err_msg = err_msg[:-1]
                err_msg += "\n{} ".format(record.rental_status_detail)
            elif record.rental_status == 'reserve': 
                err_msg = "Ruang Columbarium tersebut sudah dibooking."
                err_msg = err_msg[:-1]
                err_msg += "\n{} ".format(record.rental_status_detail)
            elif record.rental_status == 'not_available': 
                err_msg = "Ruang Columbarium tidak tersedia!"

            if err_msg:
                raise ValidationError(err_msg)
            
            if record.use_other_date:
                pickup_date = record.pickup_date_others
                return_date = record.return_date_others
            else:
                pickup_date = record.pickup_date
                return_date = record.return_date
            
            record.sale_order_id.start_period = pickup_date

            record.sale_order_id.is_rental_order = True if not record.sale_order_id.is_rental_order else record.sale_order_id.is_rental_order
            record.sale_order_id.rental_status = 'draft' if not record.sale_order_id.rental_status else record.sale_order_id.rental_status
            
            values = {
                'is_rental': True,
                'product_id': record.product_columbarium_id.id,
                'name': 'RENTAL - {product_name}\n{pickup_date} to {return_date}'.format(
                    product_name = record.product_columbarium_id.display_name,
                    pickup_date = (pickup_date+timedelta(hours=7)).strftime("%d/%m/%Y %H:%M:%S"),
                    return_date = (return_date+timedelta(hours=7)).strftime("%d/%m/%Y %H:%M:%S"),
                ),
                'product_uom_qty': 1,
                'price_unit': record.list_price,
                'tax_id': [tax.id for tax in record.product_columbarium_id.product_tmpl_id.taxes_id],
                'product_warehouse_id':record.product_columbarium_id.product_tmpl_id.default_stock_warehouse_id.id if not record.product_columbarium_id.product_tmpl_id.is_konsinyasi else int(self.env['ir.config_parameter'].sudo().get_param('custom_sale_order.kons_warehouse_id')),
                'reservation_begin': pickup_date,
                'pickup_date': pickup_date,
                'return_date': return_date,
            }
            record.sale_order_id.order_line = [(0,0,values)]
